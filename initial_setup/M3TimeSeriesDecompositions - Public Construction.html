Constructing a period of time to have a moving average, to avoid a changing economy impacting older data too harshly. Determing the moving average and square error. First and last 18 months removed, since they could not be calculated (nothing before or after them to use in the equations).

def sqErr(row):
    return (row['Public Construction'] - row['MovAvg'])**2
    
dfConstrSpnd2['MovAvgPub'] = np.NaN
dfConstrSpnd2['SqErrPub'] = np.NaN

window = 36
window = window - window % 2
for i in range(int(window/2),dfConstrSpnd2.shape[0]-int(window/2)):
    dfConstrSpnd2.loc[dfConstrSpnd2.index[i],'MovAvgPub'] = (0.5*dfConstrSpnd2.iloc[i - int(window/2)]['Public Construction'] +
    dfConstrSpnd2.iloc[i - int(window/2) + 1:i + int(window/2)]['Public Construction'].sum() +
    0.5*dfConstrSpnd2.iloc[i + int(window/2)]['Public Construction'])/float(window)
dfConstrSpnd2['SqErrPub'] = (dfConstrSpnd2['Public Construction'] - dfConstrSpnd2['MovAvgPub'])**2
dfConstrSpnd2.dropna(how='any',inplace=True)
